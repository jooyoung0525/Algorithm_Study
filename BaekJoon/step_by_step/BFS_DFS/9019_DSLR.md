###테스트케이스를 여러개 돌려야 함으로 초기화가 중요!!!

```c
#include<cstdio>
#include<cstdlib>
#include<queue>
#include<iostream>
#include<vector>

using namespace std;

queue<int> Q;
vector<int> before;

void init()
{
	while (!Q.empty())
	{
		Q.pop();
	}
	while (!before.empty())
	{
		before.pop_back();
	}
}

void split(int A,int d[])
{
	int k1, k2;
	d[0] = A / 1000;
	k1 = A % 1000;
	d[1] = k1 / 100;
	k2 = k1 % 100;
	d[2] = k2 / 10;
	d[3] = k2 % 10;
}

int left(int cur)
{
	int d[4] = { 0 };
	int temp[4] = { 0 };
	int n;
	split(cur,d);
	temp[0] = d[0];
	temp[1] = d[1];
	temp[2] = d[2];
	temp[3] = d[3];

	d[0] = temp[1];
	d[1] = temp[2];
	d[2] = temp[3];
	d[3] = temp[0];

	n = ((d[0] * 10 + d[1]) * 10 + d[2]) * 10 + d[3];

	return n;
}

int right(int cur)
{
	int d[4] = { 0 };
	int temp[4] = { 0 };
	int n;
	split(cur,d);
	temp[0] = d[0];
	temp[1] = d[1];
	temp[2] = d[2];
	temp[3] = d[3];

	d[0] = temp[3];
	d[1] = temp[0];
	d[2] = temp[1];
	d[3] = temp[2];

	n = ((d[0] * 10 + d[1]) * 10 + d[2]) * 10 + d[3];

	return n;
}



void BFS(int visit[],int before_path[],char path[],int B) {

	while (!Q.empty())
	{
		int cur = Q.front();
		Q.pop();

		if (cur == B)
		{
			before.push_back(B);

			int n = before_path[B];
			before.push_back(n);
			

			while(1) //계속반복
			{
				n = before_path[n]; //이전의 경로를 queue에 넣음
				if (n == -1) break; //-1(시작점 ==> 맨처음에 (before_path)시작값을 -1로 설정)을 만났을때 while문 나감
				before.push_back(n);
							}
			
			while (!before.empty()) //
			{
				int dir = before.back();
				before.pop_back();
				if(path[dir] != 'g' && dir != -1)
					cout << path[dir];
			}
			cout << endl;
			return;
		}

		
		int D = cur * 2;
		if (D > 9999)
		{
			D = D % 10000;
		}
		if (visit[D] == 0 && D < 10000)
		{
			Q.push(D);
			visit[D] = 1;
			before_path[D] = cur;
			path[D] = 'D';
		}

		int S;
		if(cur > 0 ) {
			 S = cur - 1;
			if (visit[S] == 0 && S > -1)  // S는 0이 될수도 있으므로 0을 포함시켜줘야함!!
			{
				Q.push(S);
				visit[S] = 1;
				before_path[S] = cur;
				path[S] = 'S';
			}
		}
		else 
		{
			S = 9999;
			if (visit[S] == 0 && S > -1) // S는 0이 될수도 있으므로 0을 포함시켜줘야함!!
			{
				Q.push(S);
				visit[S] = 1;
				before_path[S] = cur;
				path[S] = 'S';
			}
		}

		int L = left(cur);
		if (visit[L] == 0)
		{
			Q.push(L);
			visit[L] = 1;
			before_path[L] = cur;
			path[L] = 'L';
		}

		int R = right(cur);		
		if (visit[R] == 0)
		{
			Q.push(R);
			visit[R] = 1;
			before_path[R] = cur;
			path[R] = 'R';
		}
	}

}

int main()
{
	int T;
	

	cin >> T;

	for (int i = 0; i < T; i++) {

		int A, B;
		int visit[10000] = { 0, };       //테스트케이스마다 초기화해줘야해서 전역변수로선언안함.
		int before_path[10000] = { 0, };
		char path[10000] = { NULL, };

		init();

		cin >> A >> B;

		if (A != B) {
			Q.push(A);
			visit[A] = 1;
			before_path[A] = -1;
			path[A] = 'g';

			BFS(visit, before_path, path, B);
		}
	}

	//system("pause");
	return 0;
}
```
