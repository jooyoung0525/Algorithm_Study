#include<cstdio>
#include<cstdlib>


int main() {

	int N,min;

	scanf("%d", &N);

	if (N % 5 == 0)
	{
		min = N / 5;
	}
	else
	{
		for (int i = 1; i <= N / 5; i++)
		{
			if (N - (5 * i) > 0) {
				if ((N - (5 * i)) % 3 == 0)
				{
					min = i + (N - (5 * i)) / 3;
				}

				else if (N % 3 == 0)
					min = N / 3;
			}

			else
                min = -1;
		}


	}

	printf("%d", min);

  system("pause");
	return 0;
}
