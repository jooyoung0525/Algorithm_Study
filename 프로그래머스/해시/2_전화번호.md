### string find함수로 string에서 문자찾기  //틀림



```c
#include <string>
#include <vector>

using namespace std;

bool solution(vector<string> phone_book) {

	bool answer = true;

	for (int i = 0; i < phone_book.size(); i++)
	{
		for (int j = 0; j < phone_book.size(); j++)
		{
			if (i == j || i > j) continue;
			if (phone_book[j].find(phone_book[i]) != std::string::npos)
			{
				answer = false;
				break;
			}
		}
		if (answer == false)
			break;
	}
			
	return answer;
}
```
