### 풀이 1

```c
#include<iostream>
#include<vector>

using namespace std;

int N, M;
vector<int>result;

void select(int cnt, int before);

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	cin >> N >> M;
	
	select(0 ,0);

	return 0;
}

void select(int cnt, int before)
{
	if (cnt == M)
	{
		for (int i = 0; i < result.size(); i++)
		{
			cout << result[i] << " ";
		}
		cout << "\n";
		return;
	}

	for (int i = 1; i <= N; i++)
	{
		if (before > i) continue; //이전값이 현재값보다 크면 continue
		result.push_back(i);
		select(cnt + 1, i); //이전값은 앞에서 선택한 값
		result.pop_back();
	}

}
```
